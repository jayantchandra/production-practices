## ⚙️ Setup

```bash
npm init -y
npm i express dotenv cors
```

---

## 🟦 TypeScript Setup

```bash
npm i -D typescript @types/express @types/cors @types/dotenv nodemon
```

**Project structure**

* create `/src/index.ts`
* create `tsconfig.json`

**`tsconfig.json`**

```json
{
  "compilerOptions": {
    "target": "ESNext",
    "module": "NodeNext",
    "strict": true,
    "verbatimModuleSyntax": true,
    "skipLibCheck": true,
    "types": ["node"],
    "rootDir": "src",
    "outDir": "dist",
    "jsx": "react-jsx",
    "allowSyntheticDefaultImports": true,
    "esModuleInterop": true
  },
  "include": ["src/**/*"],
  "exclude": ["node_modules", "dist", ".build"]
}
```

## 📦 `package.json`

```json
"scripts": {
  "test": "echo \"Error: no test specified\" && exit 1",
  "start": "node ./dist/index.js",
  "dev:js": "nodemon ./dist/index.js",
  "dev:ts": "tsc -w",
  "build": "tsc"
},
"type": "module"
```

**Run (2 terminals)**

```bash
npm run dev:ts
npm run dev:js
```

## 🚀 Basic `index.ts`

```ts
import express, {
  type Application,
  type Request,
  type Response,
} from "express";
import "dotenv/config";

const app: Application = express();
const PORT = process.env.PORT || 4000;

app.get("/", (_req: Request, res: Response) => {
  return res.send("Server is running");
});

app.listen(PORT, () => {
  console.log(`http://localhost:${PORT}`);
});
```
